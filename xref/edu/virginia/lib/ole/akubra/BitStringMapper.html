63  			log.debug("Appending: '" + section + "' to path");
64  			path.append(section);
65  		}
66  		String uristring = path.toString() + characters;
67  		log.debug("Exiting BitStringMapper.getInternalId() with: " + uristring);
68  		return URI.create(String.valueOf(uristring));
